function loadUserData() {

} else {

habits = [];

completions = {};

}

renderHabits();

updateStats();

});

}

};

Object.keys(completions).forEach(date => {

});

await saveUserData();

};

}

function getTodayKey() {

return new Date().toISOString().split('T')[0];

}

 

function isDayComplete(date) {

const dayCompletions = completions[date] || {};

return habits.length > 0 && habits.every(habit => dayCompletions[habit.id] === true);

}

 

function isHabitCompleteOnDate(habitId, date) {

return completions[date]?.[habitId] === true;

}

 

function calculateStreakForHabit(habitId) {

const today = getTodayKey();

const yesterday = new Date(Date.now() - 86400000).toISOString().split('T')[0];

let currentStreak = 0;

let longestStreak = 0;

let longestStart = null;

let longestEnd = null;

let streakActive = false;

let checkDate = new Date();

if (isHabitCompleteOnDate(habitId, today)) {

currentStreak = 1;

streakActive = true;

} else if (isHabitCompleteOnDate(habitId, yesterday)) {

currentStreak = 1;

streakActive = true;

checkDate = new Date(Date.now() - 86400000);

}

 

if (streakActive) {

checkDate.setDate(checkDate.getDate() - 1);

while (true) {

const dateKey = checkDate.toISOString().split('T')[0];

if (isHabitCompleteOnDate(habitId, dateKey)) {

currentStreak++;

checkDate.setDate(checkDate.getDate() - 1);

} else {

break;

}

}

}

 

const dates = Object.keys(completions).sort();

let tempStreak = 0;

let tempStart = null;

 

for (const date of dates) {

if (isHabitCompleteOnDate(habitId, date)) {

if (tempStreak === 0) tempStart = date;

tempStreak++;

if (tempStreak > longestStreak) {

longestStreak = tempStreak;

longestStart = tempStart;

longestEnd = date;

}

} else {

tempStreak = 0;

tempStart = null;

}

}

 

return { currentStreak, longestStreak, longestStreakStart: longestStart, longestStreakEnd: longestEnd };

}

const yesterday = new Date(Date.now() - 86400000).toISOString().split('T')[0];

let currentStreak = 0;

} else if (isDayComplete(yesterday)) {

currentStreak = 1;

streakActive = true;

checkDate = new Date(Date.now() - 86400000);

}

 

if (streakActive) {

checkDate.setDate(checkDate.getDate() - 1);

while (true) {

const dateKey = checkDate.toISOString().split('T')[0];

if (isDayComplete(dateKey)) {

currentStreak++;

checkDate.setDate(checkDate.getDate() - 1);

} else {

break;

}

}

}

 

const dates = Object.keys(completions).sort();

let tempStreak = 0;

for (const date of dates) {

if (isDayComplete(date)) {

if (tempStreak === 0) tempStart = date;

tempStreak++;

if (tempStreak > longestStreak) {

longestStreak = tempStreak;

function updateStats() {

const overall = calculateOverallStreak();

currentStreakSpan.textContent = overall.currentStreak;

longestStreakSpan.textContent = overall.longestStreak;

if (overall.longestStreakStart) {

const startDate = new Date(overall.longestStreakStart).toLocaleDateString();

const endDate = new Date(overall.longestStreakEnd).toLocaleDateString();

longestStreakDatesSpan.textContent = `${startDate} - ${endDate}`;

longestStreakDatesSpan.classList.remove('hidden');

} else {

longestStreakDatesSpan.classList.add('hidden');

}

 

const todayComplete = isDayComplete(getTodayKey());

todayCompleteIndicator.classList.toggle('hidden', !todayComplete);

 

if (overall.currentStreak > 0) {

let message = '';

if (overall.currentStreak === 1) {

message = "Great start! Keep it going!";

} else if (overall.currentStreak < 7) {

message = `${overall.currentStreak} days strong! You're building momentum!`;

} else if (overall.currentStreak < 30) {

message = `${overall.currentStreak} day streak! You're on fire! 🔥`;

} else {

message = `Incredible! ${overall.currentStreak} days of consistency! 🏆`;

}

motivationalMessage.textContent = message;

motivationalMessage.parentElement.classList.remove('hidden');

} else {

motivationalMessage.parentElement.classList.add('hidden');

}

}

 

function updateTodayDate() {

const today = new Date();

todayDateSpan.textContent = today.toLocaleDateString('en-US', {

weekday: 'long',

month: 'long',

day: 'numeric'

});

}

</div>
